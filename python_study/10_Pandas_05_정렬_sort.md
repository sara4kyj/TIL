emp = pd.read_csv('./data/emp.csv')

#    empno  ename  deptno   sal

emp.idx

# 2. sort.values

emp.sort_values('sal')      # by 생략 가능

emp.sort_values(by='sal', ascending=False)

emp.sort_values(by=['deptno','sal'], ascending=[True,False])
